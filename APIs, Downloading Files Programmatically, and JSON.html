# List of dictionaries to build and convert to a DataFrame later
df_list = []
#a dictionary to keep track of errors recieved when trying to download the poster
image_errors = {}
#loop through title list
for i, title in enumerate(title_list):
    try:
        #movie's rank 
        rank = title_list.index(title)+1
        #query the MediaWiki API using wptools to get a movie poster URL via each page object's image attribute
        page = wptools.page(title)
        images = page.get().data['image']
        poster_url = images[0]['url']
        #file format
        image_file_format = poster_url.split('.')[-1]
        #download the poster
        r = requests.get(poster_url)
        #create an image from binary data returned by a request
        image = Image.open(BytesIO(r.content))
        #save the poster
        image.save(folder_name+'/'+str(rank)+'_'+title+'.'+image_file_format)
        df_list.append({'title':title,'ranking': int(rank),'poster_url':poster_url})
    except Exception as e:
        print(str(rank)+'_'+title+':'+str(e))
        image_errors[str(rank)+'_'+title] = images
